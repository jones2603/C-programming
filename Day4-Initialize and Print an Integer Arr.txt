Initialize and Print an Integer Array:

#include <stdio.h>

int main() {
    int numbers[] = {10, 20, 30, 40, 50};  // Initialization
    int size = sizeof(numbers) / sizeof(numbers[0]);

    printf("Array elements:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", numbers[i]);  // Printing each element
    }

    return 0;
}

Output:
Array elements:
10 20 30 40 50
