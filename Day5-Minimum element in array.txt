Minimum element in array:

#include <stdio.h>

int main() {
    int n;
    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    int min = arr[0]; // Assume first element is minimum
    for (int i = 1; i < n; i++) {
        if (arr[i] < min)
            min = arr[i];
    }

    printf("Minimum element in the array is: %d\n", min);
    return 0;
}
