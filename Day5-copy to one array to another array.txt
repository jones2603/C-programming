copy to one array to another array:

#include <stdio.h>

int main() {
    int size;
    printf("Enter the number of elements: ");
    scanf("%d", &size);

    int original[size], copy[size];

    printf("Enter %d elements:\n", size);
    for (int i = 0; i < size; i++) {
        scanf("%d", &original[i]);
        copy[i] = original[i]; // Copying element
    }

    printf("Copied array:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", copy[i]);
    }

    return 0;
}
