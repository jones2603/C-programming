
C Program: Student Marks Table Using 2D Array:

#include <stdio.h>

int main() {
    int marks[3][4];  // 3 students, 4 subjects

    // Input marks
    printf("Enter marks for 3 students in 4 subjects:\n");
    for (int i = 0; i < 3; i++) {
        printf("Student %d:\n", i + 1);
        for (int j = 0; j < 4; j++) {
            printf("  Subject %d: ", j + 1);
            scanf("%d", &marks[i][j]);
        }
    }

    // Display marks
    printf("\nMarks Table:\n");
    for (int i = 0; i < 3; i++) {
        printf("Student %d: ", i + 1);
        for (int j = 0; j < 4; j++) {
            printf("%d ", marks[i][j]);
        }
        printf("\n");
    }

    return 0;
}
